### Actualizar la llamada al checkout (OrderService):

```typescript
// src/lib/services/orders.ts
import { calculateCartTax } from "@/lib/services/tax";
import type { CartItem } from "@/store/useCartStore";

async function checkout(items: CartItem[], paymentMethod: "CASH" | "CARD") {
  const { subtotal, tax, total } = calculateCartTax(
    items.map(i => ({
      price: i.product.price,
      quantity: i.quantity,
      taxRate: i.product.taxRate,
      taxIncluded: i.product.taxIncluded,
    }))
  );

  const order = {
    id: crypto.randomUUID(),
    items: items.map(i => ({
      productId: i.product.id,
      name: i.product.name,
      price: i.product.price,
      quantity: i.quantity,
      subtotal: i.product.price * i.quantity,
      taxRate: i.product.taxRate,
      taxIncluded: i.product.taxIncluded,
    })),
    subtotal,
    tax,
    total,
    status: "COMPLETED" as const,
    paymentMethod,
    createdAt: Date.now(),
  };

  // Validar con Zod antes de enviar al Main
  OrderSchema.parse(order);

  const result = await window.electronAPI!.checkout(order);
  if (!result.success) throw new Error(result.error);
  return order;
}
```
